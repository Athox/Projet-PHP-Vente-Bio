CREATE TABLE "PROJET"."CLIENT" 
   (	"CLIENT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"NAME" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	"FIRST_NAME" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	"AGE" NUMBER(3,0) NULL,
	"SEXE" NUMBER(1,0) NULL,
	"MAIL" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	 PRIMARY KEY ("CLIENT_ID")
   );


  CREATE TABLE "PROJET"."PRODUITS" 
   (	"PRODUCT_ID" NUMBER(10,0) NOT NULL ENABLE, 
	"NAME" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	"DESCRIPTION" VARCHAR2(150 BYTE) NOT NULL ENABLE, 
	"QUANTITY_STOCK" NUMBER(7,0) NOT NULL ENABLE, 
	"KILO_PRICE" NUMBER(9,2) NOT NULL ENABLE, 
	"FRUIT" NUMBER(1,0) NOT NULL ENABLE, 
	 PRIMARY KEY ("PRODUCT_ID")
   );


CREATE TABLE "PROJET"."STATVENTE"
(	"PRODUCT_ID" NUMBER(10,0) NOT NULL ENABLE,
	"CLIENT_ID" NUMBER(*,0) NOT NULL ENABLE,
	"DATE_VENTE" DATE NOT NULL,
	"QUANTITE" NUMBER(5,0) NOT NULL,
	PRIMARY KEY ("PRODUCT_ID","CLIENT_ID")
);


CREATE TABLE "PROJET"."STATCONNEXION"
(	"CONNEXION_ID" NUMBER(10,0) NOT NULL,
	"CLIENT_ID" NUMBER(*,0) NOT NULL ENABLE,
	"DATE_CONNEXION" DATE NOT NULL,
	"TEMPS_CONNEXION" DATE NOT NULL,
	"PANIER" NUMBER(1,0) NOT NULL,
	"ACHAT" NUMBER(1,0) NOT NULL,
	PRIMARY KEY ("CONNEXION_ID","CLIENT_ID")
);


ALTER TALBE "PROJET"."STATVENTE"
ADD CONSTRAINT fk_statvente_produit_productid FOREIGN KEY ("PRODUCT_ID") REFERENCES "PROJET"."PRODUITS" ("PRODUCT_ID")
ALTER TALBE "PROJET"."STATVENTE"
ADD CONSTRAINT fk_statvente_client_clienttid FOREIGN KEY ("CLIENT_ID") REFERENCES "PROJET"."CLIENT" ("CLIENT_ID")
